CMAKE_MINIMUM_REQUIRED(VERSION 3.5 FATAL_ERROR)
PROJECT(neu.h)

SET(NeuH_MAJOR_VERSION 0)
SET(NeuH_MINOR_VERSION 1)
SET(NeuH_PATCH_VERSION 0)
SET(NeuH_VERSION ${NeuH_MAJOR_VERSION}.${NeuH_MINOR_VERSION}.${NeuH_PATCH_VERSION})

SET(NeuH_ROOT_DIR ${CMAKE_CURRENT_LIST_DIR})

CONFIGURE_FILE(${PROJECT_SOURCE_DIR}/cmake/NeuHConfigVersion.cmake.in
  "${PROJECT_BINARY_DIR}/NeuHConfigVersion.cmake" @ONLY
)

CONFIGURE_FILE(${PROJECT_SOURCE_DIR}/cmake/NeuHConfig.cmake.in
  "${PROJECT_BINARY_DIR}/NeuHConfig.cmake" @ONLY
)

EXPORT(PACKAGE NeuH)
